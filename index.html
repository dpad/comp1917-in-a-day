  1. #include <stdio.h>
  2.  
  3. // Definitions
  4. #define HT 25 // Height of the shape
  5. #define WD 50 // Width of the shape
  6.  
  7. // Function prototypes
  8. float power(float base, int index);
  9.  
  10. // Allocates a 2-dimensional array grid for storing points of a function,
  11. // then prints the grid nicely.
  12. // Grid size and function are hard-coded in. Accepting input is left as
  13. // an exercise to the reader. Suckers.
  14. int main(int argc, char *argv[]) {
  15.  
  16.     int   grid[WD][HT]; // The grid that will store the shape
  17.     int   i, j;         // Counters
  18.     float x, y;         // For storing x and y values of points
  19.  
  20.     // Store the heart shape as a grid of 1s and 0s
  21.     for (i = 0; i < HT; i++) {
  22.  
  23.         // Define the y-value for the heart function.
  24.         y = (i - HT/2) / (HT/3.0);
  25.  
  26.         for (j = 0; j < WD; j++) {
  27.  
  28.             // Define the x-value for the heart function.
  29.             x = (j - WD/2) / (WD/2.5);
  30.  
  31.             // Check if a the point lies in the heart function.
  32.             if (power((power(x, 2) + power(y, 2) - 1), 3) - 
  33.                 (power(x, 2)*power(y, 3)) <= 0) {
  34.                 // If so store a 1
  35.                 grid[j][HT-i-1] = 1;
  36.             } else {
  37.                 // Otherwise store a 0
  38.                 grid[j][HT-i-1] = 0;
  39.             }
  40.         }
  41.     }
  42.  
  43.     // Print out the grid with 1s replaced by hashes and
  44.     // 0s replaced by spaces.
  45.     for (i = 0; i < HT; i++) {
  46.         for (j = 0; j < WD; j++) {
  47.             if (grid[j][i] == 1) {
  48.                 printf("#");
  49.             } else {
  50.                 printf(" ");
  51.             }
  52.         }
  53.         printf("\n");
  54.     }
  55.  
  56.     return 0;
  57. }
  58.  
  59. // Raise a number to a given power.
  60. float power(float base, int index) {
  61.     float answer = 1;
  62.     int i;
  63.  
  64.     for (i = 0; i < index; i++) {
  65.         answer *= base;
  66.     }
  67.  
  68.     return answer;
  69. }
